175  std::string decodePath(std::string path) {
176  std::string ans = "";
177  for (int i = 1; i < path.size(); i++) {
178  if (path.at(i - 1) == path.at(i) && (path.at(i) == '.' || path.at(i) == '/')) {
179  continue;
180  }
181  ans += path.at(i - 1);
182  }
183  ans += path.at(path.size() - 1);
